import request from '@/utils/request'

/**
* ${genTable.functionName}分页查询
* @param {查询条件} data
*/
export function list${replaceDto.ModelTypeName}(query) {
  return request({
    url: '${genTable.ModuleName}/${replaceDto.ModelTypeName}/list',
    method: 'get',
    params: query,
  })
}
